    <script>
        // Show welcome message when page loads
        window.onload = function() {
            alert("Welcome! Please fill out the Student Registration Form.");
        };

        // Form validation and submission
        document.getElementById("registrationForm").addEventListener("submit", function(event) {
            event.preventDefault();
            
            // Get form values
            const fullName = document.getElementById("fullName").value.trim();
            const ticketIndex = document.getElementById("ticketIndex").value.trim();
            const email = document.getElementById("email").value.trim();
            const contact = document.getElementById("contact").value.trim();
            const course = document.getElementById("course").value;
            const permit = document.querySelector("input[name='permit']:checked");
            const includesReady = document.getElementById("ready").checked;
            const includesDance = document.getElementById("dance").checked;
            const photo = document.getElementById("photo").value;

            // Validation
            if (fullName === "" || email === "" || contact === "" || course === "" || !permit || photo === "") {
                alert("⚠️ Please fill in all required fields.");
                return;
            }

            // Email validation with regex
            const emailPattern = /^[^ ]+@[^ ]+\.[a-z]{2,3}$/;
            if (!email.match(emailPattern)) {
                alert("⚠️ Please enter a valid email address.");
                return;
            }

            // Success message
            alert("✅ Registration successful! Thank you for registering, " + fullName + ".");
            
            // Reset form (optional)
            // this.reset();
        });

        // Photo upload button functionality
        document.getElementById("uploadPhotoBtn").addEventListener("click", function() {
            document.getElementById("photo").click();
        });

        // Show file name when a file is selected
        document.getElementById("photo").addEventListener("change", function() {
            if (this.files.length > 0) {
                document.getElementById("uploadPhotoBtn").textContent = this.files[0].name;
            }
        });
